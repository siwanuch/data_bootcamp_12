# Use ggplot: visualize with Scatter plot
ggplot(data = coffee,
       mapping = aes(x=AVG_Aroma, y=AVG_Flavor,
                     color=Country.of.Origin,
                     label = coffee$Country.of.Origin)) +
  geom_point() +
  geom_text_repel(max.overlaps = 15) +
  theme_minimal() +
  theme(legend.position = "none") +
  labs(title = "Coffee Aroma vs. Flavor (by Country)",
       x = "Average Aroma",
       y = "Average Flavor")
